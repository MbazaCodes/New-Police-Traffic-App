   156→// ============================================================
   157→// HELPER: Detect search type from input
   158→// ============================================================
   159→function detectSearchType(query: string): string {
   160→  const q = query.trim();
   161→  // NIDA format: 16 digits
   162→  if (/^\d{16}$/.test(q)) return 'citizen';
   163→  // Plate format: T XXX XXX or K XXX XXX etc.
   164→  if (/^[A-Z]\s*\d{1,4}\s*[A-Z]{2,3}$/i.test(q)) return 'vehicle';
   165→  // Case number: CID-YYYY-NNN
   166→  if (/^CID-/i.test(q)) return 'case';
   167→  // PF3 form number
   168→  if (/^PF3/i.test(q)) return 'pf3';
   169→  // Accident report number
   170→  if (/^ACC/i.test(q)) return 'accident';
   171→  // Badge number: TPF-NNNN
   172→  if (/^TPF-/i.test(q)) return 'officer';
   173→  // Wanted person ID: WRT-
   174→  if (/^WRT-/i.test(q)) return 'wanted';
   175→  return 'citizen'; // default
   176→}
