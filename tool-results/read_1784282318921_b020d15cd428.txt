   480→   480→/** Tanzania plate: T NNN XYZ or T NNN XYZW — flexible */
   481→   481→export function validatePlate(v: string): ValidationResult {
   482→   482→  const clean = v.trim().toUpperCase().replace(/\s+/g, " ");
   483→   483→  if (!clean) return { valid: false, error: "Ingiza namba ya gari" };
   484→   484→  // TZ plates: T followed by 3 digits and 2-4 letters, with optional space
   485→   485→  if (!/^T\s?\d{2,4}\s?[A-Z]{2,4}$/.test(clean.replace(/\s/g, "")))
   486→   486→    return { valid: false, error: "Fomati sahihi: T 001 ABC au T123ABC" };
   487→   487→  return { valid: true, error: "" };
   488→   488→}

   491→   491→export function validateLicense(v: string): ValidationResult {

   518→   518→export function validateSerial(v: string): ValidationResult {
