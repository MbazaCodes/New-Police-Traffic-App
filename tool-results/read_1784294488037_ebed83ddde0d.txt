   245→/* ============================================================
   246→   HELPER: Badge by status
   247→   ============================================================ */
   248→function StatusBadge({ status }: { status: string }) {
   249→  const variant = status === 'Active' || status === 'Success' || status === 'Completed' || status === 'Generated'
   250→    ? 'default'
   251→    : status === 'Inactive' || status === 'Failed'
   252→      ? 'destructive'
   253→      : 'secondary';
   254→  return <Badge variant={variant} className="text-[10px] px-1.5 py-0">{status}</Badge>;
   255→}
